自己造了一个Google Chrome Extensions 谷歌浏览器的扩展应用.
删除百度音乐play.baidu.com的广告位.
目录如下:
Baidu-Music去广告:
content.js  manifest.json

------------------------------------------manifest.json------------------------------------------
{
    "name":"Clear Baidu Music..",
    "version":"1.0.0",
    "permissions":["tabs","contextMenus",""],
    "manifest_version": 2,
    "browser_action": {
        "dafault_title":"Clear Baidu Music"
    },  
    "content_scripts": [
        {   
            "matches": [""],
            "js": ["content.js"],
            "run_at":"document_end"
        }   
      ]   
}

------------------------------------------content.js------------------------------------------
function hide(){
    if (document.domain == "play.baidu.com"){
        var a = document.getElementByIdx_x_x('right-ads');
        if (a){
            a.remove();
            return true;
        }   
    }   
}

window.setTimeout(hide, 1000);
var v = false
function main(){
    if (hide()){
        if (v){
            window.clearInterval(vid);
        }   
    }   
}
var vid = window.setInterval(main, 3000);
v = true
------------------------------------------END------------------------------------------
